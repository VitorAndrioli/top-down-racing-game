45 void OBB::checkCollision(Circle* const pCircle)
46 {
47  // Performs a less costly broad check. If objects are not close enough, exits the function.
48  if (!broadCollisionCheck(pCircle)) return;
49 
50  // Calculates the distance between the centre of the objects.
51  Vector2D<double> fvCentreDistance = pCircle->getPosition() - m_fvPosition;
52  // Rotates the vector by the inverse of OBB's orientation so we can treat the OBB as an AABB.
53  fvCentreDistance.rotate(-m_fOrientation);
54 
55  // Calculates the clamp vector for the collision.
56  Vector2D<double> fvClamp;
57  if (fvCentreDistance.getX() < 0) fvClamp.setX(max(fvCentreDistance.getX(), -m_fvHalfExtents.getX()));
58  if (fvCentreDistance.getX() >= 0) fvClamp.setX(min(fvCentreDistance.getX(), m_fvHalfExtents.getX()));
59  if (fvCentreDistance.getY() < 0) fvClamp.setY(max(fvCentreDistance.getY(), -m_fvHalfExtents.getY()));
60  if (fvCentreDistance.getY() >= 0) fvClamp.setY(min(fvCentreDistance.getY(), m_fvHalfExtents.getY()));
61 
62 
63  // Calculates the distance vector between OBB's edge and circle centre.
64  Vector2D<double> fvDiff = fvCentreDistance - fvClamp;
65 
66  // If distance is smaller than the circle's radius, there is a collision.
67  // Squared values are used to avois using SQRT() function when not necessary.
68  if (fvDiff.squaredMagnitude() < (pCircle->getRadius()*pCircle->getRadius()))
69  {
70  // Calculates the overlap of the objects.
71  double fOverlap = fvDiff.magnitude() - pCircle->getRadius();
72 
73  // Rotates clamp to go back from and AABB to OBB.
74  fvClamp.rotate(m_fOrientation);
75  // Calculates collision vector subtracting the circle position from the point of contact.
76  Vector2D<double> fvCollisionNormal = (m_fvPosition + fvClamp) - pCircle->getPosition();
77  // Normalizes the collision vector.
78  fvCollisionNormal.normalize();
79 
80  Vector2D<double> fvContactPoint = m_fvPosition + fvClamp;
81  // Resolves collision
82  resolveCollision(pCircle, &fvCollisionNormal, fOverlap, &fvContactPoint);
83  }

94  if (!broadCollisionCheck(pOtherObb)) return;
95

97  array<Vector2D<double>, 4> faObbPoints;
98  array<Vector2D<double>, 4> faOtherObbPoints;
99 
100  // Gets each of the four points of OBB.
101  faObbPoints[0] = Vector2D<double>(m_fvHalfExtents.getX(), m_fvHalfExtents.getY());
102  faObbPoints[1] = Vector2D<double>(m_fvHalfExtents.getX(), -m_fvHalfExtents.getY());
103  faObbPoints[2] = Vector2D<double>(-m_fvHalfExtents.getX(), m_fvHalfExtents.getY());
104  faObbPoints[3] = Vector2D<double>(-m_fvHalfExtents.getX(), -m_fvHalfExtents.getY());
105  // Rotates all the points according to OBB orientation
106  for (array<Vector2D<double>, 4>::iterator it = faObbPoints.begin(); it != faObbPoints.end(); ++it)
107  {
108  (*it).rotate(m_fOrientation);
109  (*it) += m_fvPosition;
110  }
111 
112  // Gets each of the four points of other OBB.
113  faOtherObbPoints[0] = Vector2D<double>(pOtherObb->getHalfExtents().getX(), pOtherObb->getHalfExtents().getY());
114  faOtherObbPoints[1] = Vector2D<double>(pOtherObb->getHalfExtents().getX(), -pOtherObb->getHalfExtents().getY());
115  faOtherObbPoints[2] = Vector2D<double>(-pOtherObb->getHalfExtents().getX(), pOtherObb->getHalfExtents().getY());
116  faOtherObbPoints[3] = Vector2D<double>(-pOtherObb->getHalfExtents().getX(), -pOtherObb->getHalfExtents().getY());

118  for (array<Vector2D<double>, 4>::iterator it = faOtherObbPoints.begin(); it != faOtherObbPoints.end(); ++it)
119  {
120  (*it).rotate(pOtherObb->getOrientation());
121  (*it) += pOtherObb->getPosition();
122  }
123

125  array<Vector2D<double>, 4> axis = {
126  Vector2D<double>(cos(m_fOrientation), sin(m_fOrientation)),
127  Vector2D<double>(-sin(m_fOrientation), cos(m_fOrientation)),
128  Vector2D<double>(cos(pOtherObb->getOrientation()), sin(pOtherObb->getOrientation())),
129  Vector2D<double>(-sin(pOtherObb->getOrientation()), cos(pOtherObb->getOrientation()))
130  };
131

133  double fMinimumOverlap = 99999; // Used to find the axis with the minimum overlap.
134 
135  // Iterates trough every axis.
136  for (array<Vector2D<double>, 4>::iterator it = axis.begin(); it != axis.end(); ++it)
137  {
138  // Declares variable to store maximum and minimum points of each OBB in this axis.
139  double fObbMin = 99999, fObbMax = -99999;
140  double fOtherObbMin = 99999, fOtherObbMax = -99999;
141 
142  // Iterates trough every point of both OBBs to find the maximum and minimum values for this axis.
143  for (int j = 0; j < 4; j++)
144  {
145  // Projects the point onto axis.
146  double fObbPointProjection = (*it).dotProduct(&faObbPoints[j]);
147  // Checks if it is minimum or maximum.
148  if (fObbPointProjection < fObbMin) fObbMin = fObbPointProjection;
149  if (fObbPointProjection > fObbMax) fObbMax = fObbPointProjection;
150 
151  // Projects the point onto axis.
152  double fOtherObbPointProjection = (*it).dotProduct(&faOtherObbPoints[j]);
153  // Checks if it is minimum or maximum.
154  if (fOtherObbPointProjection < fOtherObbMin) fOtherObbMin = fOtherObbPointProjection;
155  if (fOtherObbPointProjection > fOtherObbMax) fOtherObbMax = fOtherObbPointProjection;
156  }
157 
158  // According to SAT, if there is no overlap in a single axis, there is no collision, thus we can exit the method.
159  if (fObbMax < fOtherObbMin || fOtherObbMax < fObbMin) return;
160 
161  // Gets the overlap in this axis.
162  double fOverlap = min((fObbMax - fOtherObbMin), (fOtherObbMax - fObbMin));
163 
164  // Checks if it is the minimum overlap.
165  if (fOverlap < fMinimumOverlap)
166  {
167  fMinimumOverlap = fOverlap;
168  // On SAT, the collision vector is the axis with the minimum overlap.
169  fvCollisionNormal = (*it).unitVector();
170  }
171 
172  }
173 
174  // Calculates the relative position of OBBs to know which direction the collision
175  // normal should point to. Flip collision normal if necessary.
176  Vector2D<double> fcRelativePosition = m_fvPosition - pOtherObb->getPosition();
177  if (fcRelativePosition.dotProduct(&fvCollisionNormal) < 0) fvCollisionNormal.flip();
178 
179
