89 bool Collidable::broadCollisionCheck(Collidable* const pOtherCollidable)
90 {
91  // Calculates distance of centers and sum of radii. Use squared values to avoid using sqrt().
92  double fCentreDistSquared = (m_fvPosition - pOtherCollidable->getPosition()).squaredMagnitude();
93  double fRadiiSumSquared = (m_fRadius + pOtherCollidable->getRadius()) * (m_fRadius + pOtherCollidable->getRadius());
94 
95  // Checks if the edges of the outter circles are colliding.
96  if (fCentreDistSquared - fRadiiSumSquared <= 0) return true;

108 void Collidable::resolveCollision(Collidable* pOtherCollidable, Vector2D<double>* pfvCollisionNormal, double fOverlap, Vector2D<double>* pfvContactPoint)
109 {
110  // Undoes the overlapping by moving back the other collidable.
111  pOtherCollidable->setPosition(pOtherCollidable->getPosition() + (*pfvCollisionNormal * fOverlap));
112 
113  // The collision restitution coefficient is the minimum coefficient between the collidables.
114  double fRestitution = min(m_fRestitution, pOtherCollidable->getRestitution());
115 
116 
117  // Calculates distance from centres to contact point.
118  Vector2D<double> fvDistCollidable = *pfvContactPoint - m_fvPosition;
119  Vector2D<double> fvDistOtherCollidable = *pfvContactPoint - pOtherCollidable->getPosition();
120 
121  // Calculates relative velocity.
122  //Vector2D<double> relVelocity = m_fvVelocity - otherCollidable->getVelocity();
123  Vector2D<double> relativeVelocity = (m_fvVelocity + fvDistCollidable.crossProduct(m_fAngularVelocity)) - (pOtherCollidable->getVelocity() + fvDistOtherCollidable.crossProduct(pOtherCollidable->m_fAngularVelocity));
124  // Checks if collidables are moving towards each other. If not, there is no reason to continue.
125  if (relativeVelocity.dotProduct(pfvCollisionNormal) > 0) return;
126 
127  Vector2D<double> ra = *pfvContactPoint - m_fvPosition;
128  Vector2D<double> rb = *pfvContactPoint - pOtherCollidable->getPosition();
129 
130  // Calculates impulse scalar (j).
131  //double fImpulseScalar = -(1 + fRestitution) * relativeVelocity.dotProduct(pfvCollisionNormal) / (m_fInverseMass + pOtherCollidable->getInverseMass() + (m_fInverseMomentOfInertia*pow(ra.crossProduct(pfvCollisionNormal), 2)) + (pOtherCollidable->getInverseInertia() * pow(rb.crossProduct(pfvCollisionNormal), 2)));
132  double fImpulseScalar = -(1 + fRestitution) * relativeVelocity.dotProduct(pfvCollisionNormal) / (m_fInverseMass + pOtherCollidable->getInverseMass());
133 
134  // Calculates impulses.
135  Vector2D<double> fvImpulse = *pfvCollisionNormal * fImpulseScalar;
136  applyImpulse(&fvImpulse, &fvDistCollidable);
