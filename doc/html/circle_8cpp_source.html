72  // Performs a less costly broad check. If objects are not close enough, exits the function.
73  if (!broadCollisionCheck(pObb)) return;
74

76  Vector2D<double> fvCentreDistance = m_fvPosition - pObb->getPosition();
77  // Rotates the vector by the inverse of OBB's orientation so we can treat the OBB as an AABB.
78  fvCentreDistance.rotate(-pObb->getOrientation());
79 
80  // Calculates the clamp vector for the collision.
81  Vector2D<double> fvClamp;
82  if (fvCentreDistance.getX() < 0) fvClamp.setX(max(fvCentreDistance.getX(), -pObb->getHalfExtents().getX()));
83  if (fvCentreDistance.getX() >= 0) fvClamp.setX(min(fvCentreDistance.getX(), pObb->getHalfExtents().getX()));
84  if (fvCentreDistance.getY() < 0) fvClamp.setY(max(fvCentreDistance.getY(), -pObb->getHalfExtents().getY()));

86 
87  // Calculates the distance vector between OBB's edge and circle centre.
88  Vector2D<double> fvDiff = fvCentreDistance - fvClamp;

91  // Squared values are used to avoid using SQRT() function when not necessary.
92  if (fvDiff.squaredMagnitude() < (getRadius()*getRadius()))

95  double fOverlap = fvDiff.magnitude() - getRadius();
